50  WSContext* WSContext::create(web::http::server::Request& req, web::http::server::Response& res) {
51  std::string subProtocol = req.header("sec-websocket-protocol");
52 
53  web::ws::subprotocol::WSSubProtocolPluginInterface* wSSubProtocolPluginInterface =
54  selector.select(subProtocol, web::ws::WSSubProtocol::Role::SERVER);
55 
56  web::ws::server::WSContext* wSContext = nullptr;
57 
58  if (wSSubProtocolPluginInterface != nullptr) {
59  web::ws::WSSubProtocol* wSSubProtocol = wSSubProtocolPluginInterface->create();
60 
61  if (wSSubProtocol != nullptr) {
62  wSContext = new web::ws::server::WSContext(wSSubProtocol, web::ws::WSSubProtocol::Role::SERVER);
63 
64  if (wSContext != nullptr) {
65  res.set("Upgrade", "websocket");
66  res.set("Connection", "Upgrade");
67  res.set("Sec-WebSocket-Protocol", subProtocol);
68 
69  web::ws::serverWebSocketKey(req.header("sec-websocket-key"), [&res](char* key) -> void {
70  res.set("Sec-WebSocket-Accept", key);
71  });
72 
73  res.status(101); // Switch Protocol
74  } else {
75  res.status(500);
76  }
77  } else {
78  res.status(500); // Internal Server Error
79  }
80  } else {
81  res.status(404); // Not found
82  }
83 
84  return wSContext;
