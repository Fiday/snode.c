122  void connectEvent() override {
123  errno = 0;
124  int cErrno = 0;
125 
126  socklen_t cErrnoLen = sizeof(cErrno);
127 
128  int err = getsockopt(Socket::getFd(), SOL_SOCKET, SO_ERROR, &cErrno, &cErrnoLen);
129 
130  if (err == 0) {
131  if (cErrno != EINPROGRESS) {
132  ConnectEventReceiver::disable();
133 
134  if (cErrno == 0) {
135  typename Socket::SocketAddress::SockAddr localAddress{};
136  socklen_t localAddressLength = sizeof(localAddress);
137 
138  typename Socket::SocketAddress::SockAddr remoteAddress{};
139  socklen_t remoteAddressLength = sizeof(remoteAddress);
140 
141  if (getsockname(Socket::getFd(), reinterpret_cast<sockaddr*>(&localAddress), &localAddressLength) == 0 &&
142  getpeername(Socket::getFd(), reinterpret_cast<sockaddr*>(&remoteAddress), &remoteAddressLength) == 0) {
143  ReadEventReceiver::enable();
144 
145  socketConnection =
146  new SocketConnection(onConstruct, onDestruct, onRead, onReadError, onWriteError, onDisconnect);
147 
148  socketConnection->open(Socket::getFd(), Socket::FLAGS::dontClose);
149 
150  socketConnection->setRemoteAddress(typename Socket::SocketAddress(remoteAddress));
151  socketConnection->setLocalAddress(typename Socket::SocketAddress(localAddress));
152 
153  socketConnection->ReadEventReceiver::enable();
154 
155  onError(0);
156  onConnect(socketConnection);
157  } else {
158  onError(errno);
159  }
160  } else {
161  errno = cErrno;
162  onError(errno);
163  }
164  }
165  } else {
166  ConnectEventReceiver::disable();
167  onError(errno);
168  }
169  }
