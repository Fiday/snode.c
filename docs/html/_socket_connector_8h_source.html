73  void connect(const SocketAddress& remoteAddress, const SocketAddress& bindAddress, const std::function<void(int err)>& onError) {
74  this->onError = onError;
75 
76  errno = 0;
77 
78  Socket::open(
79  [this, &bindAddress, &remoteAddress, &onError](int errnum) -> void {
80  if (errnum > 0) {
81  onError(errnum);
82  destruct();
83  } else {
84  Socket::bind(bindAddress, [this, &remoteAddress, &onError](int errnum) -> void {
85  if (errnum > 0) {
86  onError(errnum);
87  destruct();
88  } else {
89  int ret = ::connect(Socket::getFd(), &remoteAddress.getSockAddr(), remoteAddress.getSockAddrLen());
90 
91  if (ret == 0 || errno == EINPROGRESS) {
92  ConnectEventReceiver::enable(Socket::getFd());
93  } else {
94  onError(errno);
95  destruct();
96  }
97  }
98  });
99  }
100  },
101  SOCK_NONBLOCK);

107  int tryToCompleteConnect() {
108  errno = 0;
109  int cErrno = -1;
110  socklen_t cErrnoLen = sizeof(cErrno);
111 
112  int err = getsockopt(Socket::getFd(), SOL_SOCKET, SO_ERROR, &cErrno, &cErrnoLen);
113 
114  if (err == 0) {
115  if (cErrno != EINPROGRESS) {
116  if (cErrno == 0) {
117  typename SocketAddress::SockAddr localAddress{};
118  socklen_t localAddressLength = sizeof(localAddress);
119 
120  typename SocketAddress::SockAddr remoteAddress{};
121  socklen_t remoteAddressLength = sizeof(remoteAddress);
122 
123  if (getsockname(Socket::getFd(), reinterpret_cast<sockaddr*>(&localAddress), &localAddressLength) == 0 &&
124  getpeername(Socket::getFd(), reinterpret_cast<sockaddr*>(&remoteAddress), &remoteAddressLength) == 0) {
125  socketConnection =
126  new SocketConnection(onConstruct, onDestruct, onRead, onReadError, onWriteError, onDisconnect);
127 
128  socketConnection->setRemoteAddress(SocketAddress(remoteAddress));
129  socketConnection->setLocalAddress(SocketAddress(localAddress));
130 
131  socketConnection->attach(Socket::getFd());
132  socketConnection->SocketConnection::SocketReader::enable(Socket::getFd());
133 
134  SocketConnector::dontClose(true);
135  SocketConnector::ConnectEventReceiver::suspend();
136 
137  onError(0);
138  onConnect(socketConnection);
139  } else {
140  cErrno = errno;
141  onError(errno);
142  }
143  } else {
144  errno = cErrno;
145  onError(errno);
146  }
147  }
148  } else {
149  cErrno = errno;
150  onError(errno);
151  }
152 
153  return cErrno;
154  }
