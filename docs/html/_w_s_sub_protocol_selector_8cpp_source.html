46  void WSSubProtocolSelector::loadSubProtocol(const WSSubProtocolPluginInterface& subProtocol) {
47  if (subProtocol.role() == web::ws::WSSubProtocol::Role::SERVER) {
48  const auto [it, success] = serverSubprotocols.insert({subProtocol.name(), subProtocol});
49  if (!success && subProtocol.handle) {
50  dlclose(subProtocol.handle);
51  }
52  } else {
53  const auto [it, success] = clientSubprotocols.insert({subProtocol.name(), subProtocol});
54  if (!success && subProtocol.handle) {

60  void WSSubProtocolSelector::loadSubProtocols(const std::string& path) {
61  if (std::filesystem::exists(path) && std::filesystem::is_directory(path)) {
62  for (const std::filesystem::directory_entry& directoryEntry : std::filesystem::recursive_directory_iterator(path)) {
63  if (std::filesystem::is_regular_file(directoryEntry) && directoryEntry.path().extension() == ".so") {
64  void* handle = dlopen(directoryEntry.path().c_str(), RTLD_NOW | RTLD_LOCAL);
65  if (handle != nullptr) {
66  WSSubProtocolPluginInterface (*wSSubProtocolPluginInterface)(void*) =
67  reinterpret_cast<WSSubProtocolPluginInterface (*)(void*)>(dlsym(handle, "plugin"));
68 
69  loadSubProtocol(wSSubProtocolPluginInterface(handle));
70 
71  VLOG(1) << "DLOpen: success: " << directoryEntry.path().c_str();
72  } else {
73  VLOG(1) << "DLOpen: error: " << dlerror() << " - " << directoryEntry.path().c_str();
74  }
75  } else {
76  VLOG(1) << "Not a library: Ignoring direntry " << directoryEntry;
77  }
78  }
79  } else {
80  VLOG(1) << "Not a directory: Ignoring path: " << path;
81  }
82  }
