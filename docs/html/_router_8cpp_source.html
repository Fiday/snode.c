73  static bool matchFunction(const std::string& cpath, const std::string& reqpath) {
74  std::vector<std::string> explodedString = explode(cpath, '/');
75 
76  for (std::vector<std::string>::size_type i = 0; i < explodedString.size(); i++) {
77  if (explodedString[i].front() == ':') {
78  std::smatch smatch = matchResult(explodedString[i]);
79  std::string regex = "(.*)";
80 
81  if (smatch.size() > 1) {
82  if (smatch[1] != "") {
83  regex = smatch[1];
84  }
85  }
86 
87  explodedString[i] = regex;
88  }
89  }
90 
91  std::string regexPath = path_concat(explodedString);
92 
93  return std::regex_match(reqpath, std::regex(regexPath));
94  }

96  static void setParams(const std::string& cpath, Request& req) {
97  std::vector<std::string> explodedString = explode(cpath, '/');
98  std::vector<std::string> explodedReqString = explode(req.url, '/');
99 
100  for (std::vector<std::string>::size_type i = 0; i < explodedString.size() && i < explodedReqString.size(); i++) {
101  if (explodedString[i].front() == ':') {
102  std::smatch smatch = matchResult(explodedString[i]);
103  std::string regex = "(.*)";
104 
105  if (smatch.size() > 1) {
106  if (smatch[1] != "") {
107  regex = smatch[1];
108  }
109  }
110 
111  if (std::regex_match(explodedReqString[i], std::regex(regex))) {
112  std::string attributeName = smatch[0];
113  attributeName.erase(0, 1);
114  attributeName.erase((attributeName.length() - smatch[1].length()), smatch[1].length());
115 
116  req.params[attributeName] = explodedReqString[i];
117  }
118  }
119  }
120  }

127  static inline std::string path_concat(const std::string& first, const std::string& second) {
128  std::string result;
129 
130  if (first.back() == '/' && second.front() == '/') {
131  result = first + second.substr(1);
132  } else if (first.back() != '/' && second.front() != '/') {
133  result = first + '/' + second;
134  } else {
135  result = first + second;
136  }
137 
138  if (result.length() > 1 && result.back() == '/') {
139  result.pop_back();
140  }
141 
142  return result;
143  }
