34  void WSReceiver::receive(char* junk, std::size_t junkLen) {
35  uint64_t consumed = 0;
36  bool parsingError = false;
37 
38  // dumpFrame(junk, junkLen);
39 
40  while (consumed < junkLen && !parsingError) {
41  switch (parserState) {
42  case ParserState::BEGIN:
43  parserState = ParserState::OPCODE;
44  begin();
45  [[fallthrough]];
46  case ParserState::OPCODE:
47  consumed += readOpcode(junk + consumed, junkLen - consumed);
48  break;
49  case ParserState::LENGTH:
50  consumed += readLength(junk + consumed, junkLen - consumed);
51  break;
52  case ParserState::ELENGTH:
53  consumed += readELength(junk + consumed, junkLen - consumed);
54  break;
55  case ParserState::MASKINGKEY:
56  consumed += readMaskingKey(junk + consumed, junkLen - consumed);
57  break;
58  case ParserState::PAYLOAD:
59  consumed += readPayload(junk + consumed, junkLen - consumed);
60  break;
61  case ParserState::ERROR:
62  onError(errorState);
63  parsingError = true;
64  reset();
65  break;
66  };
67  }

70  uint64_t WSReceiver::readOpcode(char* junk, uint64_t junkLen) {
71  uint64_t consumed = 0;
72  if (junkLen > 0) {
73  consumed++;
74 
75  uint8_t opCodeByte = *reinterpret_cast<uint8_t*>(junk);
76 
77  fin = opCodeByte & 0b10000000;
78  opCode = opCodeByte & 0b00001111;
79 
80  if (!continuation) {
81  onMessageStart(opCode);
82  parserState = ParserState::LENGTH;
83  } else if (opCode == 0) {
84  parserState = ParserState::LENGTH;
85  } else {
86  parserState = ParserState::ERROR;
87  errorState = 1002;
88  VLOG(0) << "Error opcode in continuation frame";
89  }
90  continuation = !fin;
91  }

96  uint64_t WSReceiver::readLength(char* junk, uint64_t junkLen) {
97  uint64_t consumed = 0;
98 
99  if (junkLen > 0) {
100  uint8_t lengthByte = *reinterpret_cast<uint8_t*>(junk);
101 
102  masked = lengthByte & 0b10000000;
103  length = lengthByte & 0b01111111;
104 
105  if (length > 125) {
106  switch (length) {
107  case 126:
108  elengthNumBytes = 2;
109  break;
110  case 127:
111  elengthNumBytes = 8;
112  break;
113  }
114  parserState = ParserState::ELENGTH;
115  length = 0;
116  } else {
117  if (masked) {
118  parserState = ParserState::MASKINGKEY;
119  } else if (length > 0) {
120  parserState = ParserState::PAYLOAD;
121  } else {
122  if (fin) {
123  onMessageEnd();
124  }
125  reset();
126  }
127  }
128  consumed++;
129  }

134  uint64_t WSReceiver::readELength(char* junk, uint64_t junkLen) {
135  uint64_t consumed = 0;
136 
137  elengthNumBytesLeft = (elengthNumBytesLeft == 0) ? elengthNumBytes : elengthNumBytesLeft;
138 
139  while (consumed < junkLen && elengthNumBytesLeft > 0) {
140  length |= static_cast<uint64_t>(*reinterpret_cast<unsigned char*>(junk + consumed))
141  << (elengthNumBytes - elengthNumBytesLeft) * 8;
142 
143  consumed++;
144  elengthNumBytesLeft--;
145  }
146 
147  if (elengthNumBytesLeft == 0) {
148  switch (elengthNumBytes) {
149  case 2: {
150  length = be16toh(static_cast<uint16_t>(length));
151  } break;
152  case 8:
153  length = be64toh(length);
154  break;
155  }
156 
157  if (length & static_cast<uint64_t>(0x01) << 63) {
158  parserState = ParserState::ERROR;
159  errorState = 1004;
160  } else if (masked) {
161  parserState = ParserState::MASKINGKEY;
162  } else {
163  parserState = ParserState::PAYLOAD;
164  }
165  }
166 
167  return consumed;
168  }

197  uint64_t WSReceiver::readPayload(char* junk, uint64_t junkLen) {
198  uint64_t consumed = 0;
199 
200  if (junkLen > 0 && length - payloadRead > 0) {
201  uint64_t numBytesToRead = (junkLen <= length - payloadRead) ? junkLen : length - payloadRead;
202 
203  MaskingKey maskingKeyAsArray = {.key = htobe32(maskingKey)};
204 
205  for (uint64_t i = 0; i < numBytesToRead; i++) {
206  *(junk + i) = *(junk + i) ^ *(maskingKeyAsArray.keyAsArray + (i + payloadRead) % 4);
207  }
208 
209  onFrameData(junk, numBytesToRead);
210 
211  payloadRead += numBytesToRead;
212  consumed = numBytesToRead;
213  }
214 
215  if (payloadRead == length) {
216  if (fin) {
217  onMessageEnd();
218  }
219  reset();
220  }
221 
222  return consumed;
223  }
