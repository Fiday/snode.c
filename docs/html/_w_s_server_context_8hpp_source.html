103  template <typename WSServerProtocol>
104  void WSServerContext<WSServerProtocol>::onMessageStart(int opCode) {
105  switch (opCode) {
106  case 0x08:
107  closeReceived = true;
108  break;
109  case 0x09:
110  pingReceived = true;
111  VLOG(0) << "Ping received";
112  break;
113  case 0x0A:
114  pongReceived = true;
115  break;
116  default:
117  wSServerProtocol->onMessageStart(opCode);
118  break;
119  }

122  template <typename WSServerProtocol>
123  void WSServerContext<WSServerProtocol>::onFrameReceived(const char* junk, uint64_t junkLen) {
124  if (!closeReceived && !pingReceived && !pongReceived) {
125  std::size_t junkOffset = 0;
126 
127  do {
128  std::size_t sendJunkLen = (junkLen - junkOffset <= SIZE_MAX) ? static_cast<std::size_t>(junkLen - junkOffset) : SIZE_MAX;
129  wSServerProtocol->onFrameData(junk + junkOffset, sendJunkLen);
130  junkOffset += sendJunkLen;
131  } while (junkLen - junkOffset > 0);
132  } else {
133  // collect data for close and pong
134  }
135  }

137  template <typename WSServerProtocol>
138  void WSServerContext<WSServerProtocol>::onMessageEnd() {
139  if (closeReceived) {
140  closeReceived = false;
141  if (closeSent) { // active close
142  closeSent = false;
143  VLOG(0) << "Close confirmed from peer";
144  } else { // passive close
145  VLOG(0) << "Close request received - replying with close";
146  close();
147  }
148  socketConnection->getSocketProtocol()->close();
149  } else if (pingReceived) {
150  pingReceived = false;
151  replyPong();
152  } else if (pongReceived) {
153  pongReceived = false;
154  wSServerProtocol->onPongReceived();
155  } else {
156  wSServerProtocol->onMessageEnd();
157  }
158  }

173  template <typename WSServerProtocol>
174  void WSServerContext<WSServerProtocol>::close(uint16_t statusCode, const char* reason, std::size_t reasonLength) {
175  char* closePayload = const_cast<char*>(reason);
176  std::size_t closePayloadLength = reasonLength;
177 
178  if (statusCode != 0) {
179  closePayload = new char[reasonLength + 2];
180  *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
181  closePayloadLength += 2;
182  if (reasonLength > 0) {
183  memcpy(closePayload + 2, reason, reasonLength);
184  }
185  }
186 
187  message(8, closePayload, closePayloadLength);
188 
189  if (statusCode != 0) {
190  delete[] closePayload;
191  }
192 
193  setTimeout(CLOSE_SOCKET_TIMEOUT);
194 
195  closeSent = true;
196  }
