74  enum Parser::ParserState ResponseParser::parseHeader() {
75  for (auto& [field, value] : Parser::headers) {
76  if (field != "set-cookie") {
77  if (field == "content-length") {
78  Parser::contentLength = std::stoi(value);
79  }
80  } else {
81  std::string cookiesLine = value;
82 
83  while (!cookiesLine.empty()) {
84  std::string cookieLine;
85  std::tie(cookieLine, cookiesLine) = httputils::str_split(cookiesLine, ',');
86 
87  std::string cookieOptions;
88  std::string cookie;
89  std::tie(cookie, cookieOptions) = httputils::str_split(cookieLine, ';');
90 
91  std::string name;
92  std::string value;
93  std::tie(name, value) = httputils::str_split(cookie, '=');
94  httputils::str_trimm(name);
95  httputils::str_trimm(value);
96 
97  std::map<std::string, CookieOptions>::iterator cookieElement;
98  bool inserted;
99  std::tie(cookieElement, inserted) = cookies.insert({name, CookieOptions(value)});
100 
101  while (!cookieOptions.empty()) {
102  std::string option;
103  std::tie(option, cookieOptions) = httputils::str_split(cookieOptions, ';');
104 
105  std::string optionName;
106  std::string optionValue;
107  std::tie(optionName, optionValue) = httputils::str_split(option, '=');
108  httputils::str_trimm(optionName);
109  httputils::str_trimm(optionValue);
110 
111  cookieElement->second.setOption(optionName, optionValue);
112  }
113  }
114  }
115  }
116 
117  Parser::headers.erase("set-cookie");
118 
119  onHeader(Parser::headers, cookies);
