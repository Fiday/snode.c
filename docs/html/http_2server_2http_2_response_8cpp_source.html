39  void Response::enqueue(const char* junk, std::size_t junkLen) {
40  if (!headersSent && !sendHeaderInProgress) {
41  sendHeaderInProgress = true;
42  sendHeader();
43  sendHeaderInProgress = false;
44  headersSent = true;
45  }
46 
47  serverContext->sendResponseData(junk, junkLen);
48 
49  if (headersSent) {
50  contentSent += junkLen;
51  if (contentSent == contentLength) {
52  serverContext->responseCompleted();
53  } else if (contentSent > contentLength) {
54  serverContext->terminateConnection();
55  }
56  }
57  }

89  Response& Response::set(const std::string& field, const std::string& value, bool overwrite) {
90  if (overwrite || headers.find(field) == headers.end()) {
91  headers.insert_or_assign(field, value);
92 
93  if (field == "Content-Length") {
94  contentLength = std::stol(value);

96  connectionState = ConnectionState::Close;

146  void Response::sendHeader() {
147  enqueue("HTTP/1.1 " + std::to_string(responseStatus) + " " + StatusCode::reason(responseStatus) + "\r\n");
148  enqueue("Date: " + httputils::to_http_date() + "\r\n");
149 
150  set({{"Cache-Control", "public, max-age=0"}, {"Accept-Ranges", "bytes"}, {"X-Powered-By", "snode.c"}});
151 
152  for (auto& [field, value] : headers) {
153  enqueue(field + ": " + value + "\r\n");
154  }
155 
156  for (auto& [cookie, cookieValue] : cookies) {
157  std::string cookieString =
158  std::accumulate(cookieValue.getOptions().begin(),
159  cookieValue.getOptions().end(),
160  cookie + "=" + cookieValue.getValue(),
161  [](const std::string& str, const std::pair<const std::string&, const std::string&> option) -> std::string {
162  return str + "; " + option.first + (!option.second.empty() ? "=" + option.second : "");
163  });
164  enqueue("Set-Cookie: " + cookieString + "\r\n");
165  }
166 
167  enqueue("\r\n");
168 
169  if (headers.find("Content-Length") != headers.end()) {
170  contentLength = std::stoi(headers.find("Content-Length")->second);
171  } else {
172  contentLength = 0;
173  }
174  }
