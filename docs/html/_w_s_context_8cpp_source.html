61  void WSContext::onMessageStart(int opCode) {
62  switch (opCode) {
63  case 0x08:
64  closeReceived = true;
65  break;
66  case 0x09:
67  pingReceived = true;
68  VLOG(0) << "Ping received";
69  break;
70  case 0x0A:
71  pongReceived = true;
72  break;
73  default:
74  wSSubProtocol->onMessageStart(opCode);
75  break;
76  }
77  }

79  void WSContext::onFrameReceived(const char* junk, uint64_t junkLen) {
80  if (!closeReceived && !pingReceived && !pongReceived) {
81  std::size_t junkOffset = 0;
82 
83  do {
84  std::size_t sendJunkLen = (junkLen - junkOffset <= SIZE_MAX) ? static_cast<std::size_t>(junkLen - junkOffset) : SIZE_MAX;
85  wSSubProtocol->onMessageData(junk + junkOffset, sendJunkLen);
86  junkOffset += sendJunkLen;
87  } while (junkLen - junkOffset > 0);
88  } else {
89  pongCloseData += std::string(junk, static_cast<std::size_t>(junkLen));
90  }
91  }

93  void WSContext::onMessageEnd() {
94  if (closeReceived) {
95  closeReceived = false;
96  if (closeSent) { // active close
97  closeSent = false;
98  VLOG(0) << "Close confirmed from peer";
99  } else { // passive close
100  VLOG(0) << "Close request received - replying with close";
101  close(pongCloseData.data(), pongCloseData.length());
102  pongCloseData.clear();
103  }
104  socketConnection->getSocketProtocol()->close();
105  } else if (pingReceived) {
106  pingReceived = false;
107  replyPong(pongCloseData.data(), pongCloseData.length());
108  pongCloseData.clear();
109  } else if (pongReceived) {
110  pongReceived = false;
111  wSSubProtocol->onPongReceived();
112  } else {
113  wSSubProtocol->onMessageEnd();
114  }
115  }

144  void WSContext::close(uint16_t statusCode, const char* reason, std::size_t reasonLength) {
145  char* closePayload = const_cast<char*>(reason);
146  std::size_t closePayloadLength = reasonLength;
147 
148  if (statusCode != 0) {
149  closePayload = new char[reasonLength + 2];
150  *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
151  closePayloadLength += 2;
152  if (reasonLength > 0) {
153  memcpy(closePayload + 2, reason, reasonLength);
154  }
155  }
156 
157  close(closePayload, closePayloadLength);
158 
159  if (statusCode != 0) {
160  delete[] closePayload;
161  }
162 
163  setTimeout(CLOSE_SOCKET_TIMEOUT);
164 
165  closeSent = true;
166  }
