132  void acceptEvent() override {
133  errno = 0;
134 
135  typename Socket::SocketAddress::SockAddr remoteAddress{};
136  socklen_t remoteAddressLength = sizeof(remoteAddress);
137 
138  int scFd = -1;
139 
140  scFd = ::accept4(Socket::getFd(), reinterpret_cast<struct sockaddr*>(&remoteAddress), &remoteAddressLength, SOCK_NONBLOCK);
141 
142  if (scFd >= 0) {
143  typename Socket::SocketAddress::SockAddr localAddress{};
144  socklen_t addressLength = sizeof(localAddress);
145 
146  if (getsockname(scFd, reinterpret_cast<sockaddr*>(&localAddress), &addressLength) == 0) {
147  SocketConnection* socketConnection =
148  new SocketConnection(onConstruct, onDestruct, onRead, onReadError, onWriteError, onDisconnect);
149 
150  socketConnection->open(scFd);
151 
152  socketConnection->setRemoteAddress(typename Socket::SocketAddress(remoteAddress));
153  socketConnection->setLocalAddress(typename Socket::SocketAddress(localAddress));
154 
155  socketConnection->ReadEventReceiver::enable(scFd);
156 
157  onConnect(socketConnection);
158 
159  } else {
160  PLOG(ERROR) << "getsockname";
161  shutdown(scFd, SHUT_RDWR);
162  ::close(scFd);
163  }
164  } else if (errno != EINTR) {
165  PLOG(ERROR) << "accept";
166  }
167  }
