118  void acceptEvent() override {
119  errno = 0;
120 
121  struct sockaddr_in remoteAddress {};
122  socklen_t addrlen = sizeof(remoteAddress);
123 
124  int scFd = -1;
125 
126  scFd = ::accept4(getFd(), reinterpret_cast<struct sockaddr*>(&remoteAddress), &addrlen, SOCK_NONBLOCK);
127 
128  if (scFd >= 0) {
129  struct sockaddr_in localAddress {};
130  socklen_t addressLength = sizeof(localAddress);
131 
132  if (getsockname(scFd, reinterpret_cast<sockaddr*>(&localAddress), &addressLength) == 0) {
133  SocketConnection* socketConnection = new SocketConnection(onRead, onReadError, onWriteError, onDisconnect);
134 
135  socketConnection->open(scFd);
136 
137  socketConnection->setRemoteAddress(InetAddress(remoteAddress));
138  socketConnection->setLocalAddress(InetAddress(localAddress));
139 
140  socketConnection->ReadEventReceiver::enable();
141 
142  onConnect(socketConnection);
143  } else {
144  PLOG(ERROR) << "getsockname";
145  shutdown(scFd, SHUT_RDWR);
146  ::close(scFd);
147  }
148  } else if (errno != EINTR) {
149  PLOG(ERROR) << "accept";
