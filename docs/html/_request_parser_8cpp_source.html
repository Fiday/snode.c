65  enum Parser::ParserState RequestParser::parseStartLine(const std::string& line) {
66  enum Parser::ParserState parserState = Parser::ParserState::HEADER;
67 
68  if (!line.empty()) {
69  std::string remaining;
70 
71  std::tie(method, remaining) = httputils::str_split(line, ' ');
72  std::tie(url, httpVersion) = httputils::str_split(remaining, ' ');
73 
74  std::string queriesLine;
75  std::tie(std::ignore, queriesLine) = httputils::str_split(url, '?');
76 
77  if (!methodSupported(method)) {
78  parserState = parsingError(400, "Bad request method");
79  } else if (url.empty() || url.front() != '/') {
80  parserState = parsingError(400, "Malformed request");
81  } else {
82  std::smatch httpVersionMatch;
83  if (!std::regex_match(httpVersion, httpVersionMatch, httpVersionRegex)) {
84  parserState = parsingError(400, "Wrong protocol-version");
85  } else {
86  httpMajor = std::stoi(httpVersionMatch.str(1));
87  httpMinor = std::stoi(httpVersionMatch.str(2));
88 
89  while (!queriesLine.empty()) {
90  std::string query;
91 
92  std::tie(query, queriesLine) = httputils::str_split(queriesLine, '&');
93  queries.insert(httputils::str_split(query, '='));
94  }
95 
96  onRequest(method, url, httpVersion, httpMajor, httpMinor, queries);
97  }
98  }
99  } else {
100  parserState = parsingError(400, "Request-line empty");
101  }
102 
103  return parserState;
104  }

106  enum Parser::ParserState RequestParser::parseHeader() {
107  for (auto& [field, value] : Parser::headers) {
108  if (field != "cookie") {
109  if (field == "content-length") {
110  Parser::contentLength = std::stoi(value);
111  }
112  } else {
113  std::string cookiesLine = value;
114 
115  while (!cookiesLine.empty()) {
116  std::string cookieLine;
117  std::tie(cookieLine, cookiesLine) = httputils::str_split(cookiesLine, ',');
118 
119  while (!cookieLine.empty()) {
120  std::string cookie;
121  std::tie(cookie, cookieLine) = httputils::str_split(cookieLine, ';');
122 
123  std::string name;
124  std::string value;
125  std::tie(name, value) = httputils::str_split(cookie, '=');
126  httputils::str_trimm(name);
127  httputils::str_trimm(value);
128 
129  cookies.insert({name, value});
130  }
131  }
132  }
133  }
134 
135  Parser::headers.erase("cookie");
136 
137  onHeader(Parser::headers, cookies);
138 
139  enum Parser::ParserState parserState = Parser::ParserState::BODY;
140  if (contentLength == 0) {
141  parsingFinished();
142  parserState = ParserState::BEGIN;
143  }
144 
145  return parserState;
146  }
