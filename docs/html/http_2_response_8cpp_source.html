38  void Response::enqueue(const char* buf, size_t len) {
39  if (!headersSent && !sendHeaderInProgress) {
40  sendHeaderInProgress = true;
41  sendHeader();
42  sendHeaderInProgress = false;
43  headersSent = true;
44  }
45 
46  serverContext->sendResponseData(buf, len);
47 
48  if (headersSent) {
49  contentSent += len;
50  if (contentSent == contentLength) {
51  serverContext->responseCompleted();
52  } else if (contentSent > contentLength) {
53  serverContext->terminateConnection();
54  }
55  }
56  }

144  void Response::sendHeader() {
145  enqueue("HTTP/1.1 " + std::to_string(responseStatus) + " " + StatusCode::reason(responseStatus) + "\r\n");
146  enqueue("Date: " + httputils::to_http_date() + "\r\n");
147 
148  headers.insert({{"Cache-Control", "public, max-age=0"}, {"Accept-Ranges", "bytes"}, {"X-Powered-By", "snode.c"}});
149 
150  for (auto& [field, value] : headers) {
151  enqueue(field + ": " + value + "\r\n");
152  }
153 
154  for (auto& [cookie, cookieValue] : cookies) {
155  std::string cookieString =
156  std::accumulate(cookieValue.getOptions().begin(),
157  cookieValue.getOptions().end(),
158  cookie + "=" + cookieValue.getValue(),
159  [](const std::string& str, const std::pair<const std::string&, const std::string&> option) -> std::string {
160  return str + "; " + option.first + (!option.second.empty() ? "=" + option.second : "");
161  });
162  enqueue("Set-Cookie: " + cookieString + "\r\n");
163  }
164 
165  enqueue("\r\n");
166 
167  contentLength = std::stoi(headers.find("Content-Length")->second);
168  }
