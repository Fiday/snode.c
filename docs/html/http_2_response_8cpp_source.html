39  void Response::enqueue(const char* buf, std::size_t len) {
40  if (!headersSent && !sendHeaderInProgress) {
41  sendHeaderInProgress = true;
42  sendHeader();
43  sendHeaderInProgress = false;
44  headersSent = true;
45  }
46 
47  serverContext->sendResponseData(buf, len);
48 
49  if (headersSent) {
50  contentSent += len;
51  if (contentSent == contentLength) {
52  serverContext->responseCompleted();
53  } else if (contentSent > contentLength) {
54  serverContext->terminateConnection();
55  }
56  }
57  }

89  Response& Response::set(const std::string& field, const std::string& value, bool overwrite) {
90  if (overwrite) {
91  headers.insert_or_assign(field, value);
92  } else {
93  headers.insert({field, value});
94  }
95 
96  if (field == "Content-Length") {

147  void Response::sendHeader() {
148  enqueue("HTTP/1.1 " + std::to_string(responseStatus) + " " + StatusCode::reason(responseStatus) + "\r\n");
149  enqueue("Date: " + httputils::to_http_date() + "\r\n");
150 
151  headers.insert({{"Cache-Control", "public, max-age=0"}, {"Accept-Ranges", "bytes"}, {"X-Powered-By", "snode.c"}});
152 
153  for (auto& [field, value] : headers) {
154  enqueue(field + ": " + value + "\r\n");
155  }
156 
157  for (auto& [cookie, cookieValue] : cookies) {
158  std::string cookieString =
159  std::accumulate(cookieValue.getOptions().begin(),
160  cookieValue.getOptions().end(),
161  cookie + "=" + cookieValue.getValue(),
162  [](const std::string& str, const std::pair<const std::string&, const std::string&> option) -> std::string {
163  return str + "; " + option.first + (!option.second.empty() ? "=" + option.second : "");
164  });
165  enqueue("Set-Cookie: " + cookieString + "\r\n");
166  }
167 
168  enqueue("\r\n");
169 
170  if (headers.find("Content-Length") != headers.end()) {
171  contentLength = std::stoi(headers.find("Content-Length")->second);
172  } else {
173  contentLength = 0;
174  }
175  }
