156  requestInProgress = true;

166  template <typename Request, typename Response>
167  void ServerContext<Request, Response>::responseCompleted() {
168  RequestContext& requestContext = requestContexts.front();
169 
170  onRequestCompleted(requestContext.request, requestContext.response);
171 
172  // if 0.9 => terminate
173  // if 1.0 && (request != Keep || contentLength = -1) => terminate
174  // if 1.1 && (request == Close || contentLength = -1) => terminate
175  // if (request == Close) => terminate
176 
177  if ((requestContext.request.httpMajor == 0 && requestContext.request.httpMinor == 9) ||
178  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 0 &&
179  requestContext.request.connectionState != ConnectionState::Keep) ||
180  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 1 &&
181  requestContext.request.connectionState == ConnectionState::Close) ||
182  (requestContext.response.connectionState == ConnectionState::Close)) {
183  terminateConnection();
184  } else {
185  reset();
186 
187  requestContexts.pop_front();
188 
189  if (!requestContexts.empty() && requestContexts.front().ready) {
190  RequestContext& requestContext = requestContexts.front();
191 
192  requestInProgress = true;
193  if (requestContext.status == 0) {
194  onRequestReady(requestContext.request, requestContext.response);
195  } else {
196  requestContext.response.status(requestContext.status).send(requestContext.reason);
