62  void PipeSource::writeEvent() {

65  ssize_t ret =
66  ::write(getFd(), writeBuffer.data(), (writeBuffer.size() < MAX_SEND_JUNKSIZE) ? writeBuffer.size() : MAX_SEND_JUNKSIZE);
67 
68  if (ret > 0) {
69  writeBuffer.erase(writeBuffer.begin(), writeBuffer.begin() + ret);
70 
71  if (writeBuffer.empty()) {
72  WriteEventReceiver::suspend();
73  }
74  } else if (errno != EAGAIN && errno != EWOULDBLOCK && errno != EINTR) {
75  WriteEventReceiver::disable();
