79  void WSServerContext::onMessageStart(int opCode) {
80  closeReceived = (opCode == 0x08);
81  pingReceived = (opCode == 0x09);
82  pongReceived = (opCode == 0x0A);
83 
84  switch (opCode) {
85  case 0x08:
86  closeReceived = true;
87  VLOG(0) << "Close requested";
88  break;
89  case 0x09:
90  pingReceived = true;
91  VLOG(0) << "Ping received";
92  break;
93  case 0x0A:
94  pongReceived = true;
95  VLOG(0) << "Pong received";
96  break;
97  default:
98  _onMessageStart(this, opCode);
99  break;
100  }
101  }

103  void WSServerContext::onFrameData(const char* junk, uint64_t junkLen) {
104  if (!closeReceived && !pingReceived && !pongReceived) {
105  std::size_t junkOffset = 0;
106 
107  do {
108  std::size_t sendJunkLen = (junkLen - junkOffset <= SIZE_MAX) ? static_cast<std::size_t>(junkLen - junkOffset) : SIZE_MAX;
109  _onFrameData(this, junk + junkOffset, sendJunkLen);
110  junkOffset += sendJunkLen;
111  } while (junkLen - junkOffset > 0);
112  } else {
113  // collect data for close and pong
114  }
115  }

117  void WSServerContext::onMessageEnd() {
118  if (closeReceived) {
119  closeReceived = false;
120  if (closeSent) { // active close
121  closeSent = false;
122  VLOG(0) << "Request close";
123  } else { // passive close
124  VLOG(0) << "Closed";
125  close();
126  }
127  socketConnection->getSocketProtocol()->close();
128  } else if (pingReceived) {
129  pingReceived = false;
130  sendPong();
131  } else if (pongReceived) {
132  pongReceived = false;
133  /* Propagate connection alive to application? */
134  } else {
135  _onMessageEnd(this);
136  }
137  }

148  void WSServerContext::close(uint16_t statusCode, const char* reason, std::size_t reasonLength) {
149  char* closePayload = const_cast<char*>(reason);
150  std::size_t closePayloadLength = reasonLength;
151 
152  if (statusCode != 0) {
153  closePayload = new char[reasonLength + 2];
154  *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
155  closePayloadLength += 2;
156  if (reasonLength > 0) {
157  memcpy(closePayload + 2, reason, reasonLength);
158  }
159  }
160 
161  message(8, closePayload, closePayloadLength);
162 
163  if (statusCode != 0) {
164  delete[] closePayload;
165  }
166 
167  closeSent = true;
168  }
