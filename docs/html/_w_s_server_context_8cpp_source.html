95  void WSServerContext::close(uint16_t statusCode, const char* reason, std::size_t reasonLength) {
96  char* closePayload = const_cast<char*>(reason);
97  std::size_t closePayloadLength = reasonLength;
98 
99  if (statusCode != 0) {
100  closePayload = new char[reasonLength + 2];
101  *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
102  closePayloadLength += 2;
103  if (reasonLength > 0) {
104  memcpy(closePayload + 2, reason, reasonLength);
105  }
106  }
107 
108  message(8, closePayload, closePayloadLength);
109 
110  if (statusCode != 0) {
111  delete[] closePayload;
112  }
113 
114  closeSent = true;
