38  void Response::sendFile(const std::string& file, const std::function<void(int err)>& onError) {
39  std::string absolutFileName = file;
40 
41  if (std::filesystem::exists(absolutFileName)) {
42  std::error_code ec;
43  absolutFileName = std::filesystem::canonical(absolutFileName);
44 
45  if (std::filesystem::is_regular_file(absolutFileName, ec) && !ec) {
46  headers.insert({{"Content-Type", MimeTypes::contentType(absolutFileName)},
47  {"Last-Modified", httputils::file_mod_http_date(absolutFileName)}});
48  headers.insert_or_assign("Content-Length", std::to_string(std::filesystem::file_size(absolutFileName)));
49 
50  fileReader = FileReader::read(
51  absolutFileName,
52  [this](char* data, int length) -> void {
53  enqueue(data, length);
54  },
55  [this, onError](int err) -> void {
56  if (onError) {
57  onError(err);
58  }
59  if (err != 0) {
60  serverContext->terminateConnection();
61  }
62  });
63  } else {
64  responseStatus = 403;
65  errno = EACCES;
66  serverContext->terminateConnection();
67  if (onError) {
68  onError(EACCES);
69  }
70  end();
71  }
72  } else {
73  responseStatus = 404;
74  errno = ENOENT;
75  serverContext->terminateConnection();
76  if (onError) {
