53 
54  SocketContext* context = nullptr;

56  web::websocket::server::SubProtocolFactory* subProtocolFactory = subProtocolFactorySelector.select(subProtocolName);
57 
58  if (subProtocolFactory != nullptr) {
59  SubProtocol* subProtocol = subProtocolFactory->create();
60 
61  if (subProtocol != nullptr) {

63 
64  if (context != nullptr) {
65  response->set("Upgrade", "websocket");

67  response->set("Sec-WebSocket-Protocol", subProtocolName);
68 
69  base64::serverWebSocketKey(request->header("sec-websocket-key"), [&response = this->response](char* key) -> void {
70  response->set("Sec-WebSocket-Accept", key);
71  });
72 
73  response->status(101).end(); // Switch Protocol
74  } else {
75  delete subProtocol;
76  response->status(500).end(); // Internal Server Error
77  }
