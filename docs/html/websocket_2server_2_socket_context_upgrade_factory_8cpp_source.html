46  SocketContext* SocketContextUpgradeFactory::create(net::socket::stream::SocketConnection* socketConnection) {
47  std::string subProtocolName = request->header("sec-websocket-protocol");
48 
49  SocketContext* socketContext = nullptr;
50 
51  web::websocket::server::SubProtocolFactory* subProtocolFactory = SubProtocolFactorySelector::instance()->select(subProtocolName);
52 
53  if (subProtocolFactory != nullptr) {
54  SubProtocol* subProtocol = subProtocolFactory->createSubProtocol();
55 
56  if (subProtocol != nullptr) {
57  socketContext = new SocketContext(socketConnection, subProtocol);
58 
59  if (socketContext != nullptr) {
60  incRefCount();
61 
62  socketContext->setSocketContextUpgradeFactory(this);
63  subProtocol->setSocketContext(socketContext);
64  subProtocol->setSubProtocolFactory(subProtocolFactory);
65 
66  response->set("Upgrade", "websocket");
67  response->set("Connection", "Upgrade");
68  response->set("Sec-WebSocket-Protocol", subProtocolName);
69  response->set("Sec-WebSocket-Accept", base64::serverWebSocketKey(request->header("sec-websocket-key")));
70 
71  response->status(101).end(); // Switch Protocol
72  } else {
73  subProtocolFactory->deleteSubProtocol(subProtocol);
74  response->set("Connection", "close");
75  response->status(500).end(); // Internal Server Error
76  }
77  } else {
78  response->set("Connection", "close");
79  response->status(404).end(); // Not Found
80  }
81  } else {
82  response->set("Connection", "close");
83  response->status(404).end(); // Not Found
84  }
85

87  checkRefCount();
88  }
89

91  }
