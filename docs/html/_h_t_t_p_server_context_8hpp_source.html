137  template <typename Request, typename Response>
138  void HTTPServerContext<Request, Response>::requestParsed() {
139  if (!requestInProgress) {
140  RequestContext& requestContext = requestContexts.front();
141 
142  requestInProgress = true;
143  if (requestContext.status == 0) {
144  if (((requestContext.request.connectionState == ConnectionState::Close) ||
145  (requestContext.request.httpMajor == 0 && requestContext.request.httpMinor == 9) ||
146  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 0 &&
147  requestContext.request.connectionState != ConnectionState::Keep) ||
148  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 1 &&
149  requestContext.request.connectionState == ConnectionState::Close))) {
150  requestContext.response.set("Connection", "close");
151  } else {
152  requestContext.response.set("Connection", "keep-alive");
153  }
154 
155  onRequestReady(requestContext.request, requestContext.response);
156  } else {
157  requestContext.response.status(requestContext.status).send(requestContext.reason);
158  terminateConnection();
159  }
160  }
161  }

163  template <typename Request, typename Response>
164  void HTTPServerContext<Request, Response>::responseCompleted() {
165  RequestContext& requestContext = requestContexts.front();
166 
167  // if 0.9 => terminate
168  // if 1.0 && (request != Keep || contentLength = -1) => terminate
169  // if 1.1 && (request == Close || contentLength = -1) => terminate
170  // if (request == Close) => terminate
171 
172  if ((requestContext.request.httpMajor == 0 && requestContext.request.httpMinor == 9) ||
173  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 0 &&
174  requestContext.request.connectionState != ConnectionState::Keep) ||
175  (requestContext.request.httpMajor == 1 && requestContext.request.httpMinor == 1 &&
176  requestContext.request.connectionState == ConnectionState::Close) ||
177  (requestContext.response.connectionState == ConnectionState::Close)) {
178  terminateConnection();
179  } else {
180  reset();
181 
182  requestContexts.pop_front();
183 
184  if (!requestContexts.empty() && requestContexts.front().ready) {
185  RequestContext& requestContext = requestContexts.front();
186 
187  requestInProgress = true;
188  if (requestContext.status == 0) {
189  onRequestReady(requestContext.request, requestContext.response);
190  } else {
191  requestContext.response.status(requestContext.status).send(requestContext.reason);
192  terminateConnection();
193  }
194  }
195  }
