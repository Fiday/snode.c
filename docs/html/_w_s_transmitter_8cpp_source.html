52  void WSTransmitter::send(bool end, uint8_t opCode, char* message, std::size_t messageLength, uint32_t messageKey) {
53  std::size_t messageOffset = 0;
54 
55  do {
56  std::size_t sendMessageLength =
57  (messageLength - messageOffset <= WSMAXFRAMEPAYLOADLENGTH) ? messageLength - messageOffset : WSMAXFRAMEPAYLOADLENGTH;
58 
59  bool fin = sendMessageLength == messageLength - messageOffset;
60 
61  sendFrame(fin && end, opCode, messageKey, message + messageOffset, sendMessageLength);
62 
63  messageOffset += sendMessageLength;
64 
65  opCode = 0; // continuation
66  } while (messageLength - messageOffset > 0);
67  }

129  void WSTransmitter::sendFrame(bool fin, uint8_t opCode, uint32_t maskingKey, char* payload, uint64_t payloadLength) {
130  uint64_t length = 0;
131 
132  if (payloadLength < 126) {
133  length = payloadLength;
134  } else if (payloadLength < 0x10000) {
135  length = 126;
136  } else {
137  length = 127;
138  }
139 
140  char header[2];
141  header[0] = static_cast<char>((fin ? 0b10000000 : 0) | opCode);
142  header[1] = static_cast<char>(((maskingKey > 0) ? 0b10000000 : 0) | length);
143 
144  sendFrameData(header, 2);
145 
146  switch (length) {
147  case 126:
148  sendFrameData(htobe16(static_cast<uint16_t>(payloadLength)));
149  break;
150  case 127:
151  sendFrameData(htobe64(static_cast<uint64_t>(payloadLength)));
152  break;
153  }
154 
155  if (maskingKey > 0) {
156  sendFrameData(htobe32(maskingKey));
157  }
158 
159  union MaskingKey {
160  uint32_t keyAsValue;
161  char keyAsBytes[4];
162  };
163 
164  MaskingKey maskingKeyAsArray = {.keyAsValue = htobe32(maskingKey)};
165 
166  for (uint64_t i = 0; i < payloadLength; i++) {
167  *(payload + i) = *(payload + i) ^ *(maskingKeyAsArray.keyAsBytes + i % 4);
168  }
169 
170  sendFrameData(payload, payloadLength);
171  }
