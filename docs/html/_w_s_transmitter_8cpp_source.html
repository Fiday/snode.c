49  void WSTransmitter::send(bool end, uint8_t opCode, const char* message, std::size_t messageLength, uint32_t messageKey) {
50  std::size_t messageOffset = 0;
51 
52  do {
53  std::size_t sendMessageLength =
54  (messageLength - messageOffset <= WSMAXFRAMEPAYLOADLENGTH) ? messageLength - messageOffset : WSMAXFRAMEPAYLOADLENGTH;
55 
56  bool fin = sendMessageLength == messageLength - messageOffset;
57 
58  sendFrame(fin && end, opCode, messageKey, message + messageOffset, sendMessageLength);
59 
60  messageOffset += sendMessageLength;
61 
62  opCode = 0; // continuation
63  } while (messageLength - messageOffset > 0);
64  }

66  void WSTransmitter::sendFrame(bool fin, uint8_t opCode, uint32_t maskingKey, const char* payload, uint64_t payloadLength) {
67  uint64_t frameLength = 2 + payloadLength;
68  uint8_t opCodeOffset = 0;
69  uint8_t lengthOffset = 1;
70  uint8_t eLengthOffset = 2;
71  uint8_t maskingKeyOffset = 2;
72  uint8_t payloadOffset = 2;
73 
74  if (maskingKey > 0) {
75  frameLength += 4;
76  payloadOffset += 4;
77  }
78 
79  char* frame = nullptr;
80  uint64_t length = 0;
81 
82  if (payloadLength < 126) {
83  frame = new char[frameLength];
84  length = payloadLength;
85  } else if (payloadLength < 0x10000) {
86  frameLength += 2;
87  maskingKeyOffset += 2;
88  payloadOffset += 2;
89  frame = new char[frameLength];
90  *reinterpret_cast<uint16_t*>(frame + eLengthOffset) = htobe16(static_cast<uint16_t>(payloadLength));
91  length = 126;
92  } else {
93  frameLength += 8;
94  maskingKeyOffset += 8;
95  payloadOffset += 8;
96  frame = new char[frameLength];
97  *reinterpret_cast<uint64_t*>(frame + eLengthOffset) = htobe64(payloadLength);
98  length = 127;
99  }
100 
101  if (maskingKey > 0) {
102  *reinterpret_cast<uint32_t*>(frame + maskingKeyOffset) = htobe32(maskingKey);
103  }
104 
105  *reinterpret_cast<uint8_t*>(frame + opCodeOffset) = static_cast<uint8_t>((fin ? 0b10000000 : 0) | opCode);
106  *reinterpret_cast<uint8_t*>(frame + lengthOffset) = static_cast<uint8_t>(((maskingKey > 0) ? 0b10000000 : 0) | length);
107 
108  MaskingKey maskingKeyAsArray = {.key = htobe32(maskingKey)};
109 
110  for (uint64_t i = 0; i < payloadLength; i++) {
111  *(frame + payloadOffset + i) = *(payload + i) ^ *(maskingKeyAsArray.keyAsArray + i % 4);
112  }
113 
114  // dumpFrame(frame, frameLength);
115 
116  onFrameReady(frame, frameLength);
117 
118  delete[] frame;
119  }
