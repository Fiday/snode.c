52  void WSTransmitter::send(bool end, uint8_t opCode, const char* message, std::size_t messageLength, uint32_t messageKey) {
53  std::size_t messageOffset = 0;
54 
55  do {
56  std::size_t sendMessageLength =
57  (messageLength - messageOffset <= WSMAXFRAMEPAYLOADLENGTH) ? messageLength - messageOffset : WSMAXFRAMEPAYLOADLENGTH;
58 
59  bool fin = sendMessageLength == messageLength - messageOffset;
60 
61  sendFrame(fin && end, opCode, messageKey, message + messageOffset, sendMessageLength);
62 
63  messageOffset += sendMessageLength;
64 
65  opCode = 0; // continuation
66  } while (messageLength - messageOffset > 0);
67  }

69  void WSTransmitter::sendFrame(bool fin, uint8_t opCode, uint32_t maskingKey, const char* payload, uint64_t payloadLength) {
70  uint64_t frameLength = 2 + payloadLength;
71  uint8_t opCodeOffset = 0;
72  uint8_t lengthOffset = 1;
73  uint8_t eLengthOffset = 2;
74  uint8_t maskingKeyOffset = 2;
75  uint8_t payloadOffset = 2;
76 
77  if (maskingKey > 0) {
78  frameLength += 4;
79  payloadOffset += 4;
80  }
81 
82  char* frame = nullptr;
83  uint64_t length = 0;
84 
85  if (payloadLength < 126) {
86  frame = new char[frameLength];
87  length = payloadLength;
88  } else if (payloadLength < 0x10000) {
89  frameLength += 2;
90  maskingKeyOffset += 2;
91  payloadOffset += 2;
92  frame = new char[frameLength];
93  *reinterpret_cast<uint16_t*>(frame + eLengthOffset) = htobe16(static_cast<uint16_t>(payloadLength));
94  length = 126;
95  } else {
96  frameLength += 8;
97  maskingKeyOffset += 8;
98  payloadOffset += 8;
99  frame = new char[frameLength];
100  *reinterpret_cast<uint64_t*>(frame + eLengthOffset) = htobe64(payloadLength);
101  length = 127;
102  }
103 
104  if (maskingKey > 0) {
105  *reinterpret_cast<uint32_t*>(frame + maskingKeyOffset) = htobe32(maskingKey);
106  }
107 
108  *reinterpret_cast<uint8_t*>(frame + opCodeOffset) = static_cast<uint8_t>((fin ? 0b10000000 : 0) | opCode);
109  *reinterpret_cast<uint8_t*>(frame + lengthOffset) = static_cast<uint8_t>(((maskingKey > 0) ? 0b10000000 : 0) | length);
110 
111  MaskingKey maskingKeyAsArray = {.key = htobe32(maskingKey)};
112 
113  for (uint64_t i = 0; i < payloadLength; i++) {
114  *(frame + payloadOffset + i) = *(payload + i) ^ *(maskingKeyAsArray.keyAsArray + i % 4);
115  }
116 
117  // dumpFrame(frame, frameLength);
118 
119  onFrameReady(frame, frameLength);
120 
121  delete[] frame;
122  }
