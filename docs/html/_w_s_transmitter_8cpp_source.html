56  void WSTransmitter::send(bool end, uint8_t opCode, const char* message, std::size_t messageLength) {
57  std::size_t messageOffset = 0;
58 
59  do {
60  std::size_t sendMessageLength =
61  (messageLength - messageOffset <= WSMAXFRAMEPAYLOADLENGTH) ? messageLength - messageOffset : WSMAXFRAMEPAYLOADLENGTH;
62 
63  bool fin = sendMessageLength == messageLength - messageOffset;
64 
65  sendFrame(fin && end, opCode, message + messageOffset, sendMessageLength);
66 
67  messageOffset += sendMessageLength;
68 
69  opCode = 0; // continuation
70  } while (messageLength - messageOffset > 0);

133  void WSTransmitter::sendFrame(bool fin, uint8_t opCode, const char* payload, uint64_t payloadLength) {
134  uint64_t length = 0;
135 
136  if (payloadLength < 126) {
137  length = payloadLength;
138  } else if (payloadLength < 0x10000) {
139  length = 126;
140  } else {
141  length = 127;
142  }
143 
144  char header[2];
145  header[0] = static_cast<char>((fin ? 0b10000000 : 0) | opCode);
146 
147  header[1] = static_cast<char>((masking ? 0b10000000 : 0) | length);
148 
149  sendFrameData(header, 2);
150 
151  switch (length) {
152  case 126:
153  sendFrameData(htobe16(static_cast<uint16_t>(payloadLength)));
154  break;
155  case 127:
156  sendFrameData(htobe64(static_cast<uint64_t>(payloadLength)));
157  break;
158  }
159 
160  union MaskingKey {
161  uint32_t keyAsValue;
162  char keyAsBytes[4];
163  };
164 
165  MaskingKey maskingKeyAsArray = {.keyAsValue = distribution(generator)};
166 
167  if (masking) {
168  sendFrameData(htobe32(maskingKeyAsArray.keyAsValue));
169 
170  for (uint64_t i = 0; i < payloadLength; i++) {
171  *(const_cast<char*>(payload) + i) = *(payload + i) ^ *(maskingKeyAsArray.keyAsBytes + i % 4);
172  }
173  }
174 
175  sendFrameData(payload, payloadLength);
176 
177  if (masking) {
178  for (uint64_t i = 0; i < payloadLength; i++) {
179  *(const_cast<char*>(payload) + i) = *(payload + i) ^ *(maskingKeyAsArray.keyAsBytes + i % 4);
