60  void SocketContext::onMessageStart(int opCode) {
61  switch (opCode) {
62  case 0x08:
63  closeReceived = true;
64  break;
65  case 0x09:
66  pingReceived = true;
67  VLOG(0) << "Ping received";
68  break;
69  case 0x0A:
70  pongReceived = true;
71  break;
72  default:
73  subProtocol->onMessageStart(opCode);
74  break;
75  }
76  }

78  void SocketContext::onFrameReceived(const char* junk, uint64_t junkLen) {
79  if (!closeReceived && !pingReceived && !pongReceived) {
80  std::size_t junkOffset = 0;
81 
82  do {
83  std::size_t sendJunkLen = (junkLen - junkOffset <= SIZE_MAX) ? static_cast<std::size_t>(junkLen - junkOffset) : SIZE_MAX;
84  subProtocol->onMessageData(junk + junkOffset, sendJunkLen);
85  junkOffset += sendJunkLen;
86  } while (junkLen - junkOffset > 0);
87  } else {
88  pongCloseData += std::string(junk, static_cast<std::size_t>(junkLen));
89  }
90  }

92  void SocketContext::onMessageEnd() {
93  if (closeReceived) {
94  closeReceived = false;
95  if (closeSent) { // active close
96  closeSent = false;
97  VLOG(0) << "Close confirmed from peer";
98  } else { // passive close
99  VLOG(0) << "Close request received - replying with close";
100  close(pongCloseData.data(), pongCloseData.length());
101  pongCloseData.clear();
102  }
103  close();
104  } else if (pingReceived) {
105  pingReceived = false;
106  replyPong(pongCloseData.data(), pongCloseData.length());
107  pongCloseData.clear();
108  } else if (pongReceived) {
109  pongReceived = false;
110  subProtocol->onPongReceived();
111  } else {
112  subProtocol->onMessageEnd();
113  }
114  }

145  void SocketContext::close(uint16_t statusCode, const char* reason, std::size_t reasonLength) {
146  char* closePayload = const_cast<char*>(reason);
147  std::size_t closePayloadLength = reasonLength;
148 
149  if (statusCode != 0) {
150  closePayload = new char[reasonLength + 2];
151  *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
152  closePayloadLength += 2;
153  if (reasonLength > 0) {
154  memcpy(closePayload + 2, reason, reasonLength);
155  }
156  }
157 
158  close(closePayload, closePayloadLength);
159 
160  if (statusCode != 0) {
161  delete[] closePayload;
162  }
163 
164  setTimeout(CLOSE_SOCKET_TIMEOUT);
165 
166  closeSent = true;
167  }
