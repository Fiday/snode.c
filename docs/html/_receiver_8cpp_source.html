72 
73  if (consumed > 0) {
74  uint8_t opCodeByte = static_cast<uint8_t>(byte);
75 
76  fin = opCodeByte & 0b10000000;

95  std::size_t Receiver::readLength() {
96  char byte = 0;
97  std::size_t consumed = readFrameData(&byte, 1);
98 
99  if (consumed > 0) {
100  uint8_t lengthByte = static_cast<uint8_t>(byte);
101 
102  masked = lengthByte & 0b10000000;
103  length = lengthByte & 0b01111111;
104 
105  if (length > 125) {

133  std::size_t Receiver::readELength() {
134  std::size_t consumed = 0;
135 
136  elengthNumBytesLeft = (elengthNumBytesLeft == 0) ? elengthNumBytes : elengthNumBytesLeft;
137 
138  std::size_t elengthJunkLen = readFrameData(elengthJunk, elengthNumBytesLeft);
139 
140  while (elengthJunkLen - consumed > 0) {
141  length |= static_cast<uint64_t>(*reinterpret_cast<unsigned char*>(elengthJunk + consumed))
142  << (elengthNumBytes - elengthNumBytesLeft) * 8;
143 
144  consumed++;
145  elengthNumBytesLeft--;
146  }
147 
148  if (elengthNumBytesLeft == 0) {
149  switch (elengthNumBytes) {
150  case 2: {
151  length = be16toh(static_cast<uint16_t>(length));
152  } break;
153  case 8:
154  length = be64toh(length);
155  break;
156  }
157 
158  if (length & static_cast<uint64_t>(0x01) << 63) {
159  parserState = ParserState::ERROR;
160  errorState = 1004;
161  } else if (masked) {
162  parserState = ParserState::MASKINGKEY;
163  } else {
164  parserState = ParserState::PAYLOAD;
165  }
166  }
167 
168  return consumed;

200  std::size_t Receiver::readPayload() {
201  std::size_t consumed = 0;
202 
203  std::size_t payloadJunkLen = (MAX_PAYLOAD_JUNK_LEN <= length - payloadRead) ? static_cast<std::size_t>(MAX_PAYLOAD_JUNK_LEN)
204  : static_cast<std::size_t>(length - payloadRead);
205 
206  payloadJunkLen = readFrameData(payloadJunk, payloadJunkLen);
207 
208  if (payloadJunkLen > 0) {
209  MaskingKey maskingKeyAsArray = {.key = htobe32(maskingKey)};
210 
211  for (std::size_t i = 0; i < payloadJunkLen; i++) {
212  *(payloadJunk + i) = *(payloadJunk + i) ^ *(maskingKeyAsArray.keyAsArray + (i + payloadRead) % 4);
213  }
214 
215  onFrameReceived(payloadJunk, payloadJunkLen);
216 
217  payloadRead += payloadJunkLen;
218  consumed = payloadJunkLen;
219  }
220 
221  if (payloadRead == length) {
222  if (fin) {
223  onMessageEnd();
224  }
225  reset();
226  }
227 
228  return consumed;
229  }
