90  void sendClose(uint16_t statusCode = 1000, const char* reason = nullptr, std::size_t reasonLength = 0) {
91  char* closePayload = const_cast<char*>(reason);
92  std::size_t closePayloadLength = reasonLength;
93 
94  if (statusCode != 0) {
95  closePayload = new char[reasonLength + 2];
96  *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
97  closePayloadLength += 2;
98  if (reasonLength > 0) {
99  memcpy(closePayload + 2, reason, reasonLength);
100  }
101  }
102 
103  sendClose(closePayload, closePayloadLength);
104 
105  if (statusCode != 0) {
106  delete[] closePayload;
107  }
108 
109  setTimeout(CLOSE_SOCKET_TIMEOUT);
110 
111  closeSent = true;
112  }

146  void onFrameReceived(const char* junk, uint64_t junkLen) override {
147  switch (opCodeReceived) {
148  case OpCode::CLOSE:
149  [[fallthrough]];
150  case OpCode::PING:
151  [[fallthrough]];
152  case OpCode::PONG:
153  pongCloseData += std::string(junk, static_cast<std::size_t>(junkLen));
154  break;
155  default:
156  std::size_t junkOffset = 0;
157 
158  do {
159  std::size_t sendJunkLen =
160  (junkLen - junkOffset <= SIZE_MAX) ? static_cast<std::size_t>(junkLen - junkOffset) : SIZE_MAX;
161  subProtocol->onMessageData(junk + junkOffset, sendJunkLen);
162  junkOffset += sendJunkLen;
163  } while (junkLen - junkOffset > 0);
164  break;
165  }
166  }
167  void onMessageEnd() override {
168  switch (opCodeReceived) {
169  case OpCode::CLOSE:
170  if (closeSent) { // active close
171  closeSent = false;
172  VLOG(0) << "Close confirmed from peer";
173  } else { // passive close
174  VLOG(0) << "Close request received - replying with close";
175  sendClose(pongCloseData.data(), pongCloseData.length());
176  pongCloseData.clear();
177  }
178  break;
179  case OpCode::PING:
180  sendPong(pongCloseData.data(), pongCloseData.length());
181  pongCloseData.clear();
182  break;
183  case OpCode::PONG:
184  subProtocol->onPongReceived();
185  break;
186  default:
187  subProtocol->onMessageEnd();
188  break;
189  }
190  }
