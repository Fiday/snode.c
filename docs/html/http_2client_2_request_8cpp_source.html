116  void Request::enqueue(const char* junk, std::size_t junkLen) {
117  if (!headersSent && !sendHeaderInProgress) {
118  sendHeaderInProgress = true;
119  sendHeader();
120  sendHeaderInProgress = false;
121  headersSent = true;
122  }
123 
124  clientContext->sendRequestData(junk, junkLen);
125 
126  if (headersSent) {
127  contentSent += junkLen;
128  if (contentSent == contentLength) {
129  clientContext->requestCompleted();
130  } else if (contentSent > contentLength) {
131  clientContext->terminateConnection();
132  }
133  }
134  }

140  void Request::sendHeader() {
141  std::string httpVersion = "HTTP/" + std::to_string(httpMajor) + "." + std::to_string(httpMinor);
142 
143  std::string queryString = "";
144  if (queries.size() > 0) {
145  queryString += "?";
146  for (auto& [key, value] : queries) {
147  queryString += httputils::url_encode(key) + "=" + httputils::url_encode(value) + "&";
148  }
149  queryString.pop_back();
150  }
151 
152  enqueue(method + " " + url + queryString + " " + httpVersion + "\r\n");
153 
154  enqueue("Host: " + host + "\r\n");
155  enqueue("Date: " + httputils::to_http_date() + "\r\n");
156 
157  headers.insert({{"Cache-Control", "public, max-age=0"}, {"Accept-Ranges", "bytes"}, {"X-Powered-By", "snode.c"}});
158 
159  for (auto& [field, value] : headers) {
160  enqueue(field + ":" + value + "\r\n");
161  }
162 
163  if (contentLength != 0) {
164  enqueue("Content-Length: " + std::to_string(contentLength) + "\r\n");
165  }
166 
167  if (connectionState == ConnectionState::Keep) {
168  enqueue("Connection: keep-alive\r\n");
169  } else {
170  enqueue("Connection: close\r\n");
171  }
172 
173  for (auto& [name, value] : cookies) {
174  enqueue("Cookie:" + name + "=" + value + "\r\n");
175  }
176 
177  enqueue("\r\n");
178 
179  if (headers.find("Content-Length") != headers.end()) {
180  contentLength = std::stoi(headers.find("Content-Length")->second);
181  } else {
182  contentLength = 0;
183  }
184  }
