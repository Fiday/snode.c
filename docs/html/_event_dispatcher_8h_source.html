62  void enable(EventReceiver* eventReceiver, int fd) {
63  if (contains(disabledEventReceiver[fd], eventReceiver)) {
64  // same tick
65  disabledEventReceiver[fd].remove(eventReceiver);
66  } else if (!eventReceiver->isEnabled() &&
67  (!enabledEventReceiver.contains(fd) || !contains(enabledEventReceiver[fd], eventReceiver))) {
68  // normal
69  enabledEventReceiver[fd].push_back(eventReceiver);
70  eventReceiver->enabled(fd);

72  LOG(WARNING) << "EventReceiver double enable";
73  }

133  struct timeval observeEnabledEvents() {
134  struct timeval nextTimeout = {LONG_MAX, 0};
135 
136  for (auto [fd, eventReceivers] : enabledEventReceiver) {
137  for (EventReceiver* eventReceiver : eventReceivers) {
138  observedEventReceiver[fd].push_front(eventReceiver);
139  if (!eventReceiver->isSuspended()) {
140  fdSet.set(fd);

154  struct timeval dispatchActiveEvents(struct timeval currentTime) {
155  struct timeval nextInactivityTimeout {
156  LONG_MAX, 0
157  };
158 
159  for (const auto& [fd, eventReceivers] : observedEventReceiver) {
160  EventReceiver* eventReceiver = eventReceivers.front();
161  struct timeval maxInactivity = eventReceiver->getTimeout();
162  if (fdSet.isSet(fd)) {
163  eventCounter++;
164  eventReceiver->dispatchEvent();
165  eventReceiver->setLastTriggered(currentTime);
166  nextInactivityTimeout = std::min(nextInactivityTimeout, maxInactivity);
167  } else {
168  struct timeval inactivity = currentTime - eventReceiver->getLastTriggered();
169  if (inactivity >= maxInactivity) {
170  eventReceiver->timeoutEvent();
171  eventReceiver->disable();
172  } else {
173  nextInactivityTimeout = std::min(maxInactivity - inactivity, nextInactivityTimeout);
174  }
175  }
176  }
177 
178  return nextInactivityTimeout;

181  void unobserveDisabledEvents() {
182  for (auto [fd, eventReceivers] : disabledEventReceiver) {
183  for (EventReceiver* eventReceiver : eventReceivers) {
184  observedEventReceiver[fd].remove(eventReceiver);
185  if (observedEventReceiver[fd].empty() || observedEventReceiver[fd].front()->isSuspended()) {
186  if (observedEventReceiver[fd].empty()) {
187  observedEventReceiver.erase(fd);
188  }
189  fdSet.clr(fd, true);
190  } else {
191  fdSet.set(fd);
192  observedEventReceiver[fd].front()->setLastTriggered({time(nullptr), 0});
193  }
194  eventReceiver->disabled();
195  if (eventReceiver->observationCounter == 0) {
196  unobservedEventReceiver.push_back(eventReceiver);
197  }
198  eventReceiver->fd = -1;
199  }
200  }
201  disabledEventReceiver.clear();
202  }
