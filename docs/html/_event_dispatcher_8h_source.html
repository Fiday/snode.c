115  long dispatchActiveEvents(const fd_set& fdSet, int& counter, time_t currentTime) {
116  long nextInactivityTimeout = LONG_MAX;
117 
118  for (const auto& [fd, eventReceivers] : observedEventReceiver) {
119  EventReceiver* eventReceiver = eventReceivers.front();
120  long maxInactivity = eventReceiver->getTimeout();
121  if (FD_ISSET(fd, &fdSet)) {
122  eventCounter++;
123  counter--;
124  dispatchEventTo(eventReceiver);
125  eventReceiver->setLastTriggered(currentTime);
126  nextInactivityTimeout = std::min(nextInactivityTimeout, maxInactivity);
127  } else {
128  long inactivity = currentTime - eventReceiver->getLastTriggered();
129  if (inactivity >= maxInactivity) {
130  eventReceiver->disable();
131  } else {
132  nextInactivityTimeout = std::min(maxInactivity - inactivity, nextInactivityTimeout);
133  }
134  }
135  }
136 
137  return nextInactivityTimeout;
138  }
