83  template <typename SubProtocolT>
84  void SocketContext<SubProtocolT>::onFrameReceived(const char* junk, uint64_t junkLen) {
85  switch (opCodeReceived) {
86  case OpCode::CLOSE:
87  [[fallthrough]];
88  case OpCode::PING:
89  [[fallthrough]];
90  case OpCode::PONG:
91  pongCloseData += std::string(junk, static_cast<std::size_t>(junkLen));
92  break;
93  default:
94  std::size_t junkOffset = 0;
95 
96  do {
97  std::size_t sendJunkLen =
98  (junkLen - junkOffset <= SIZE_MAX) ? static_cast<std::size_t>(junkLen - junkOffset) : SIZE_MAX;
99  subProtocol->onMessageData(junk + junkOffset, sendJunkLen);
100  junkOffset += sendJunkLen;
101  } while (junkLen - junkOffset > 0);
102  break;
103  }
104  }

106  template <typename SubProtocolT>
107  void SocketContext<SubProtocolT>::onMessageEnd() {
108  switch (opCodeReceived) {
109  case OpCode::CLOSE:
110  if (closeSent) { // active close
111  closeSent = false;
112  VLOG(0) << "Close confirmed from peer";
113  } else { // passive close
114  VLOG(0) << "Close request received - replying with close";
115  sendClose(pongCloseData.data(), pongCloseData.length());
116  pongCloseData.clear();
117  }
118  break;
119  case OpCode::PING:
120  sendPong(pongCloseData.data(), pongCloseData.length());
121  pongCloseData.clear();
122  break;
123  case OpCode::PONG:
124  subProtocol->onPongReceived();
125  break;
126  default:
127  subProtocol->onMessageEnd();
128  break;
129  }
130  }

163  template <typename SubProtocolT>
164  void SocketContext<SubProtocolT>::sendClose(uint16_t statusCode, const char* reason, std::size_t reasonLength) {
165  char* closePayload = const_cast<char*>(reason);
166  std::size_t closePayloadLength = reasonLength;
167 
168  if (statusCode != 0) {
169  closePayload = new char[reasonLength + 2];
170  *reinterpret_cast<uint16_t*>(closePayload) = htobe16(statusCode);
171  closePayloadLength += 2;
172  if (reasonLength > 0) {
173  memcpy(closePayload + 2, reason, reasonLength);
174  }
175  }
176 
177  sendClose(closePayload, closePayloadLength);
178 
179  if (statusCode != 0) {
180  delete[] closePayload;
181  }
182 
183  setTimeout(CLOSE_SOCKET_TIMEOUT);
184 
185  closeSent = true;
186  }
